CREATE DATABASE UTILES

--Crear Las tablas

USE UTILES
CREATE TABLE GOMAS
(
	ID_GOMA INT PRIMARY KEY IDENTITY(1000,1),
	PRECIO FLOAT NOT NULL,
	MARCA VARCHAR(50) NULL,
	TIPO VARCHAR(50) NOT NULL
)
CREATE TABLE LAPICES
(
	ID_LAPIZ INT PRIMARY KEY IDENTITY(2000,1),
	PRECIO FLOAT NOT NULL,
	MARCA VARCHAR(50) NULL,
	COLOR VARCHAR(50) NOT NULL
)
CREATE TABLE SACAPUNTAS
(
	ID_SPUNTA INT PRIMARY KEY IDENTITY(3000,1),
	PRECIO FLOAT NOT NULL,
	MARCA VARCHAR(50) NULL,
	MATERIAL VARCHAR(50) NOT NULL
)
